cmake_minimum_required(VERSION 3.20)
project(TestLogger VERSION 0.1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)
FetchContent_Declare(
    fmt
    GIT_REPOSITORY https://github.com/fmtlib/fmt.git
    GIT_TAG 10.2.1
)
FetchContent_MakeAvailable(fmt)

add_library(LoggerCore STATIC
    core/logger/Logger.cpp
    core/logger/Logger.h
)

target_include_directories(LoggerCore PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(LoggerCore PUBLIC fmt::fmt)

add_executable(test_logger test_logger.cpp)
target_link_libraries(test_logger LoggerCore)
